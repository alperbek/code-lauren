draw: { ?x ?y
  circle-width: 35
  circle-height: 35

  clear-screen()
  draw-oval(add(220 x)
            add(200 y)
            circle-width
            circle-height
            "filled"
            "blue")
}

angle: 0

forever {
  radius-of-circle-orbit: 100
  circle-angle-change: 0.2

  angle: add(angle circle-angle-change)

  draw(multiply(radius-of-circle-orbit cosine(angle))
       multiply(radius-of-circle-orbit sine(angle)))
}
